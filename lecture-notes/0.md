data Natural:
  | zero
  | succ(pred :: Natural)
end

fun num-to-nat(num :: Number) -> Natural:
  doc: ``` a multiline

  ask: 
    | num < 0 then: raise("cannot be nat")
    | num == 0 then: zero
    | otherwise: succ(num-to-nat(num - 1))
  end
where:
  num-to-nat(0) is zero
  num-to-nat(1) is succ(zero)
  num-to-nat(4) is succ(succ(succ(succ(zero))))
end

fun nat-to-num(n :: Natural) -> Number:
  cases (Natural) n:
    | zero => 0
    | succ(pred) => 1 + nat-to-num(pred)
  end
where:
  nat-to-num(zero) is 0
  nat-to-num(succ(zero)) is 1
  nat-to-num(succ(succ(succ(succ(zero))))) is 4
end

fun sub1(nat :: Natural) -> Natural:
  cases (Natural) nat:
    | zero => raise("cannot sub1 from zero")
    | succ(pred) => pred 
  end
where:
  sub1(zero) raises "cannot sub1 from zero"
  sub1(succ(zero)) is zero
  sub1(succ(succ(zero))) is succ(zero)
end

fun add-slow(l :: Natural, r :: Natural) -> Natural:
  ask:
    | (l == zero) and (r == zero) then: zero
    | l == zero then: r
    | r == zero then: l
    | otherwise: succ(succ(add-slow(sub1(l), sub1(r))))
  end
where:
  add-slow(zero, zero) is zero
  add-slow(succ(zero), zero) is succ(zero)
  add-slow(zero, succ(zero)) is succ(zero)
  add-slow(succ(zero), succ(zero)) is succ(succ(zero))
end

fun add(l :: Natural, r :: Natural) -> Natural:
  cases (Natural) l:
    | zero => r
    | succ(pred) => succ(add(pred, r)) 
  end
where:
  add(zero, zero) is zero
  add(succ(zero), zero) is succ(zero)
  add(zero, succ(zero)) is succ(zero)
  add(succ(zero), succ(zero)) is succ(succ(zero))
end

fun fib(n :: Natural) -> Natural:
  cases (Natural) n:
    | zero => zero
    | succ(p1) => 
      cases (Natural) p1:
        | zero => succ(zero)
        | succ(p2) => add(fib(p1), fib(p2))
      end 
  end
where:
  fib(zero) is zero
  fib(succ(zero)) is succ(zero)
  fib(succ(succ(zero))) is succ(zero)
  fib(num-to-nat(10)) is num-to-nat(55)
end
